windows 脚本技巧

md5sum.exe md5sum.exe
输出格式为 md5value filename

现在想在bat脚本中比较两个文件的md5值

for /f "tokens=1" %%i in ('"md5sum file1"') do set old_md5=%%i
for /f "tokens=1" %%i in ('"md5sum file2"') do set new_md5=%%i
if %old_md5% == %new_md5% echo same md5
